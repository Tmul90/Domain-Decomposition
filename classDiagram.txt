' Interfaces
' ====================
interface IWeapon {
	+Shoot()
}

interface IThrowable {
	+Use()
}

interface IGadget {
	+Deploy()
}

interface IExplodable { 
	+Explode() 
}

interface INonExplodingThrowable {
	+Hit()
}

interface IDrillerEquipment {
	+Action()
}

' Attributes
' ==================== 
abstract class Attribute {
	+name: String
}

class DamageAttribute extends Attribute {
	+damageAmount: Float
	+type: String ' "Kinetic", "Explosive", "Fire", "Frost" '
}

class RadiusAttribute extends Attribute {
	+value: Float
}

class FuseAttribute extends Attribute {
	+time: Float
}

class CooldownAttribute extends Attribute {
	+time: Float
}

class DurationAttribute extends Attribute {
	+time: Float
}

class StatusEffectAttribute extends Attribute {
	+effect: String
	+duration: Float
}

' Equipment Base Classes
' ====================
abstract class Equipment {
	+name: String
}

class Gadget extends Equipment implements IGadget
abstract class Weapon extends Equipment implements IWeapon
class PrimaryWeapon extends Weapon
class SecondaryWeapon extends Weapon

abstract class Throwable extends Equipment implements IThrowable

' Driller Equipment
' ==================== 
class Flamethrower extends Equipment implements IDrillerEquipment
class WaveCooker extends Equipment implements IDrillerEquipment
class PowerDrills extends Gadget implements IDrillerEquipment
class SatchelCharge extends Gadget implements IDrillerEquipment, IThrowable, IExplodable
class ImpactAxe extends Equipment implements INonExplodingThrowable, IThrowable

Flamethrower *-- DamageAttribute : FireDamage(25)
Flamethrower *-- StatusEffectAttribute : Ignite(5s)

WaveCooker *-- DamageAttribute : PhysicalDamage
WaveCooker *-- StatusEffectAttribute : Stun

ImpactAxe *-- DamageAttribute : PhysicalDamage
ImpactAxe *-- StatusEffectAttribute : Knockback

PowerDrills *-- StatusEffectAttribute : DrillSpeedBoost

SatchelCharge *-- DamageAttribute : ExplosiveDamage
SatchelCharge *-- RadiusAttribute : BlastRadius
SatchelCharge *-- FuseAttribute : FuseTime

' Classes
' ==================== 
abstract class ClassBase {
	+name: String
	+equip(e: Equipment)
}

class Driller extends ClassBase
class Engineer extends ClassBase
class Gunner extends ClassBase
class Scout extends ClassBase
